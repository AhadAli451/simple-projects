  user_input1 = int(input())
  user_input2 = int(input())

  print(f"Tow number sum is complete= {total}")


  
  user_input1 = int(input())
  user_input2 = int(input())

  print(f"The difference of the two numbers is complete= {total}")

  user_input1 = int(input())
